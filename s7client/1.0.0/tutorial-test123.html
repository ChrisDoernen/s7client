    Read DB

    

const {S7Client} = require('../src/index');

// PLC Connection Settings
const plcSettings = {
  name: "MyPLC",
  host: 'localhost',
  port: 102,
  rack: 0,
  slot: 2
};

// DBA to read
let dbNr = 1000;
let dbVars = [
  {
    ident: "Byte 2",
    type: "BYTE",
    start: 2
  },
  {
    ident: "Wort 2",
    type: 'WORD',
    start: 2
  },
  {
    ident: "Wort 3",
    type: 'WORD',
    start: 4
  },
  {
    ident: "Wort 5",
    type: 'WORD',
    start: 8
  },
  {
    ident: "Byte 9",
    type: "BYTE",
    start: 9
  },
  {
    ident: "Bool",
    type: "BOOL",
    start: 1,
    bit: 2
  }
];


let client = new S7Client(plcSettings);
client.on('error', console.error);

client.connect()
  .then(cpuinfo => {
    console.log('Connection successful');
    console.log(cpuinfo);
    console.time('ReadDB');

    return client.readDB(dbNr, dbVars);
  })
  .then(res => {
    console.timeEnd('ReadDB');
    res.forEach(v => {
      console.log(`${v.ident}: ${v.value.toString(16)}`);
    });
  })
  .then(() => client.disconnect())
  .catch(console.error);
