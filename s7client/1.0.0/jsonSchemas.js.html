    const datatypes = require('./datatypes');
/**
 * @module jsonSchemas
 */

/**
 * JSON Schema for a Var
 */
module.exports.var = {
  type: "object",
  required: ["type", "area", "dbnr", "start"],
  properties: {
    type: {
      enum: Object.keys(datatypes),
    },
    area: {
      enum: ['pe', 'db'],
    },
    dbnr: {
      type: "integer",
    },
    start: {
      type: "integer",
    },
    bit: {
      type: "integer",
      minimum: 0,
      maximum: 7
    },
  }
};

/**
 * JSON Schema for Array of Vars
 */
module.exports.vars = {
  type: "array",
  maxItems: 15, // Must fit in one PDU
  items: module.exports.var
};

/**
 * JSON Schema for a single Var from a DB-Var
 */
module.exports.dbvar = {
  type: "object",
  required: ["type", "start"],
  properties: {
    type: {
      enum: Object.keys(datatypes),
    },
    start: {
      type: "integer",
    },
    bit: {
      type: "integer",
      minimum: 0,
      maximum: 7
    },
  }
};

/**
 * JSON Schema for Array of DB-Vars
 */
module.exports.dbvars = {
  type: "array",
  items: module.exports.dbvar
};
